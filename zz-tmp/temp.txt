/**
 * Explodes a consolidated file filter regex into individual line-by-line regex strings.
 * @param {string} regexStr - The original regex string to explode.
 * @returns {string[]} An array of individual regex strings.
 */
function explodeRegexFilter(regexStr) {
    // 1. Separate the non-extension part from the extensions part
    const [noExtensionPart, extensionsPart] = regexStr.split('|');

    if (!extensionsPart) {
        return [regexStr];
    }

    // 2. Extract the pipe-separated extensions from inside the group ( ?:ext1|ext2... )
    const extensionsMatch = extensionsPart.match(/\((?:\?:)?([^)]+)\)/);

    if (!extensionsMatch) {
        return [regexStr];
    }

    // 3. Split the extensions and rebuild them individually
    const extensions = extensionsMatch[1].split('|');
    const individualExtensions = extensions.map(ext => `.*\\.${ext}$`);

    // 4. Return the non-extension rule combined with the individual extension rules
    return [noExtensionPart, ...individualExtensions];
}

// === Example Usage ===
const inputRegex = "^[^.]+$|.*\\.(?:log|tmp|lock|zip|tar|png|jpg|gif|pyc|vsix)$";
const result = explodeRegexFilter(inputRegex);

console.log(result);
